    echo   'comparing timestep0000000000.h5'                                                                             &&
    h5diff -v $dir/timestep0000000000.h5               ${CMAKE_CURRENT_SOURCE_DIR}/refdata/$dir/timestep0000000000.h5    &&
    echo   'comparing timestep0000000556.h5'                                                                             &&

  done 
")
